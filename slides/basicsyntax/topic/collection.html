map { ... }


val numbers = setOf(1, 2, 3)
println(numbers.map { it * 3 })

filter { ... }


val numbers = listOf("one", "two", "three", "four")
val longerThan3 = numbers.filter { it.length > 3 }
println(longerThan3)
